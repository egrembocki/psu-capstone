 31class SDR:
 32    """Python counterpart of NuPIC's SparseDistributedRepresentation.
 33
 34    The instance maintains dense, sparse, and coordinate views that lazily
 35    materialise from whichever representation is currently authoritative.
 36    Callbacks can be registered to observe mutations or destruction events.
 37
 38    Attributes:
 39        __dimensions: Shape of the SDR as a list of ints.
 40        __size: Total number of bits in the SDR.
 41        _dense: Backing dense bit vector representing active elements.
 42        _sparse: Cached list of active indices in sparse form.
 43        _coordinates: Cached coordinates for each active bit broken per dimension.
 44        _dense_valid: Flag indicating whether the dense buffer is authoritative.
 45        _sparse_valid: Flag indicating whether the sparse buffer is authoritative.
 46        _coordinates_valid: Flag indicating whether the coordinate cache is valid.
 47        __callbacks: Registered change callbacks invoked after value updates.
 48        __destroy_callbacks: Callbacks invoked during ``destroy``.
 49    """

 51    def __init__(self, dimensions: Sequence[int]) -> None:
 52        """Create a new SDR with the given dimensions.
 53
 54        Args:
 55            dimensions: Iterable defining the length of each SDR dimension.
 56
 57        Raises:
 58            AssertionError: If no dimensions are provided.
 59        """
 60        self.__dimensions: List[int] = [int(dim) for dim in dimensions]
 61        nta_check(len(self.__dimensions) > 0, "SDR must have at least one dimension.")
 62
 63        self.__size: int = prod(int(dim) for dim in self.__dimensions)
 64
 65        self._dense: sdr_dense_t = [elem_dense(0)] * int(self.__size)
 66        self._sparse: sdr_sparse_t = []
 67        self._coordinates: sdr_coordinate_t = [[] for _ in self.__dimensions]

 70        self._sparse_valid = False
 71        self._coordinates_valid = False

 96    def set_dense_inplace(self) -> None:
 97        """Mark the dense buffer as authoritative after in-place edits.
 98
 99        Verifies the dense array size, coerces values into ``elem_dense``, and
100        invalidates cached sparse/coordinate views before notifying callbacks.
101        """
102
103        nta_assert(
104            len(self._dense) == int(self.__size),
105            "Dense buffer size does not match SDR size.",
106        )
107
108        self._dense = [elem_dense(int(val)) for val in self._dense]
109
110        self.clear()
111        self._dense_valid = True
112        self.do_callbacks()

114    def set_sparse_inplace(self) -> None:
115        """Mark the sparse buffer as authoritative after in-place edits.
116
117        Ensures sparse indices are sorted, unique, and within bounds, then
118        refreshes cached dense/coordinate views and triggers callbacks.
119        """
120
121        nta_assert(
122            all(int(self._sparse[i]) <= int(self._sparse[i + 1])
123                for i in range(len(self._sparse) - 1)),
124            "Sparse data must be sorted!",
125        )
126        if self._sparse:
127            nta_assert(
128                int(self._sparse[-1]) < int(self.__size),
129                "Sparse index out of bounds!",
130            )
131
132        previous = None
133        for idx in self._sparse:
134            nta_assert(
135                previous is None or int(idx) != int(previous),
136                "Sparse data must not contain duplicates!",
137            )
138            previous = idx
139
140        self._sparse = [elem_sparse(int(val)) for val in self._sparse]
141
142        self.clear()
143        self._sparse_valid = True
144        self.do_callbacks()

146    def set_coordinates_inplace(self) -> None:
147        """Mark the coordinate buffers as authoritative after edits.
148
149        Validates that every dimension shares the same active count, that
150        indices respect per-dimension bounds, and that caches reflect the new
151        canonical coordinate ordering.
152        """
153
154        nta_assert(
155            len(self._coordinates) == len(self.__dimensions),
156            "Coordinate data must match SDR dimensionality!",
157        )
158
159        expected_length = len(self._coordinates[0]) if self._coordinates else 0
160        for dim_index, coord_vec in enumerate(self._coordinates):
161            nta_assert(
162                len(coord_vec) == expected_length,
163                "All coordinate vectors must share the same length!",
164            )
165            limit = int(self.__dimensions[dim_index])
166            for idx in coord_vec:
167                nta_assert(
168                    int(idx) < limit,
169                    "Coordinate index out of bounds!",
170                )
171
172        self._coordinates = [
173            [elem_sparse(int(idx)) for idx in coord_vec] for coord_vec in self._coordinates
174        ]
175
176        self.clear()
177        self._coordinates_valid = True
178        self.do_callbacks()

242    def set_dense(self, dense: Iterable[int]) -> None:
243        """Replace contents with a dense iterable after validating its length."""
244        dense_list = list(dense)
245        nta_assert(
246            len(dense_list) == int(self.__size),
247            "Input dense array size does not match SDR size.",
248        )

250        temp = [elem_dense(int(val)) for val in dense_list]

252        self.set_dense_inplace()

254    def get_dense(self) -> sdr_dense_t:
255        """Return the dense representation, materialising it from sparse data if required."""
256        if not self._dense_valid:
257            if not self._sparse_valid:
258                self.get_sparse()
259            self._dense = [elem_dense(0)] * int(self.__size)
260            for index in self._sparse:
261                self._dense[int(index)] = elem_dense(1)
262            self._dense_valid = True
263        return self._dense

265    def at_byte(self, coordinates: Sequence[int]) -> int:
266        """Return the value stored at the provided multidimensional coordinate.
267
268        Performs bounds checking, computes the flattened index, and returns
269        the stored dense byte without mutating caches.
270        """
271        nta_assert(
272            len(coordinates) == len(self.__dimensions),
273            "Number of coordinates must match dimensionality.",
274        )
275
276        flat_index = 0
277        stride = 1
278        for dim_size, coord in zip(reversed(self.__dimensions), reversed(coordinates)):
279            nta_assert(int(coord) < int(dim_size), "Coordinate out of bounds.")
280            flat_index += int(coord) * stride
281            stride *= int(dim_size)
282        return self.get_dense()[flat_index]

284    def set_sparse(self, sparse: Iterable[int]) -> None:
285        """Replace the SDR contents with sparse indices and recompute caches."""
286        self._sparse = [elem_sparse(int(idx)) for idx in sparse]
287        self.set_sparse_inplace()

289    def get_sparse(self) -> sdr_sparse_t:
290        """Return sparse indices, creating them from dense or coordinate caches as needed."""
291        if not self._sparse_valid:
292            if self._dense_valid:
293                self._sparse = [
294                    elem_sparse(idx) for idx, value in enumerate(self._dense) if int(value) != 0
295                ]
296            elif self._coordinates_valid:
297                self._sparse = []
298                length = len(self._coordinates[0]) if self._coordinates else 0
299                for idx in range(length):
300                    flat_index = 0
301                    stride = 1
302                    for dim_idx in range(len(self.__dimensions) - 1, -1, -1):
303                        coord = self._coordinates[dim_idx][idx]
304                        flat_index += int(coord) * stride
305                        stride *= int(self.__dimensions[dim_idx])
306                    self._sparse.append(elem_sparse(flat_index))
307            else:
308                self._sparse = []
309            self._sparse.sort(key=int)
310            self._sparse_valid = True
311        return self._sparse
312
313    def set_coordinates(self, coordinates: Iterable[Iterable[int]]) -> None:
314        """Replace the SDR contents with explicit coordinates per dimension."""
315        self._coordinates = [
316            [elem_sparse(int(idx)) for idx in coord_vec] for coord_vec in coordinates
317        ]
318        self.set_coordinates_inplace()

320    def get_coordinates(self) -> sdr_coordinate_t:
321        """Return coordinate lists, deriving them from the sparse view when stale."""
322        if not self._coordinates_valid:
323            for coord in self._coordinates:
324                coord.clear()
325
326            for index in self.get_sparse():
327                flat_index = int(index)
328                for dim in range(len(self.__dimensions) - 1, 0, -1):
329                    dim_size = int(self.__dimensions[dim])
330                    self._coordinates[dim].append(flat_index % dim_size)
331                    flat_index //= dim_size
332                self._coordinates[0].append(flat_index)
333
334            self._coordinates_valid = True
335        return self._coordinates

337    def set_sdr(self, other: "SDR") -> None:
338        """Copy shape and active bits from another SDR, reshaping if necessary."""
339        other_dims = other.get_dimensions()
340        if not self.__dimensions:
341            self.__dimensions = [int(dim) for dim in other_dims]
342            self.__size = prod(int(dim) for dim in self.__dimensions)
343        else:
344            self.reshape(other_dims)
345        self.set_sparse(int(idx) for idx in other.get_sparse())

382    def intersection(self, sdrs: List["SDR"]) -> None:
383        """Compute the bitwise intersection of multiple SDRs and store it in-place.
384
385        Args:
386            sdrs: Collection of SDRs to intersect with this instance.
387
388        Raises:
389            AssertionError: If fewer than two SDRs are provided or if any SDR
390            has incompatible dimensions.
391        """
392        nta_check(len(sdrs) >= 2, "Intersection requires at least two SDRs.")
393
394        inputs = list(sdrs)
395        inplace = False
396        i = 0
397        while i < len(inputs):
398            sdr = inputs[i]
399            nta_check(sdr is not None, INPUT_SDR_NONE_MSG)
400            nta_check(
401                sdr.get_dimensions() == self.__dimensions,
402                "All SDRs must share dimensions for intersection.",
403            )
404            if sdr is self:
405                inplace = True
406                inputs[i] = inputs[-1]
407                inputs.pop()
408                continue
409            i += 1
410
411        if inplace:
412            dense_buffer = self.get_dense()
413        else:
414            dense_buffer = [elem_dense(int(val)) for val in inputs[-1].get_dense()]
415            self._dense = dense_buffer
416            inputs.pop()
417
418        for sdr in inputs:
419            data = sdr.get_dense()
420            for idx, val in enumerate(data):
421                dense_buffer[idx] = elem_dense(1 if int(dense_buffer[idx]) and int(val) else 0)
422
423        self.set_dense_inplace()

425    def _validate_concatenate_inputs(self, inputs: List["SDR"], axis_index: int) -> int:
426        """Validate concatenate inputs and return the combined size along the chosen axis."""
427        concat_axis_size = 0
428        for sdr in inputs:
429            nta_check(sdr is not None, "Input SDR cannot be None.")
430            dims = sdr.get_dimensions()
431            nta_check(
432                len(dims) == len(self.__dimensions),
433                "Input dimensionality mismatch.",
434            )
435            for dim_idx, (dim_in, dim_self) in enumerate(zip(dims, self.__dimensions)):
436                if dim_idx == axis_index:
437                    concat_axis_size += int(dim_in)
438                else:
439                    nta_check(
440                        int(dim_in) == int(dim_self),
441                        "All non-axis dimensions must match for concatenate.",
442                    )
443        return concat_axis_size

488    def concatenate(self, inputs: List["SDR"], axis: int) -> None:
489        """Concatenate SDRs along a chosen axis, writing the dense result into this instance.
490
491        Args:
492            inputs: SDRs to concatenate into this SDR.
493            axis: Axis index along which to concatenate.
494
495        Raises:
496            AssertionError: If fewer than two inputs are provided, if the axis
497            is invalid, or if input dimensions are incompatible with ``self``.
498        """
499        nta_check(len(inputs) >= 2, "Not enough inputs to concatenate.")
500
501        axis_index = int(axis)
502        nta_check(0 <= axis_index < len(self.__dimensions), "Axis out of bounds.")
503
504        concat_axis_size = self._validate_concatenate_inputs(inputs, axis_index)
505
506        nta_check(
507            concat_axis_size == int(self.__dimensions[axis_index]),
508            "Concatenation axis dimensions do not sum to output size.",
509        )
510
511        buffers = [list(sdr.get_dense()) for sdr in inputs]
512        row_lengths: List[int] = []
513        for sdr in inputs:
514            row = 1
515            dims = sdr.get_dimensions()
516            for dim_idx in range(axis_index, len(dims)):
517                row *= int(dims[dim_idx])
518            row_lengths.append(row)
519
520        total_size = int(self.__size)
521        self._dense = [elem_dense(0)] * total_size
522        positions = [0] * len(inputs)
523        out_pos = 0
524
525        while out_pos < total_size:
526            for buf_idx, buf in enumerate(buffers):
527                row = row_lengths[buf_idx]
528                start = positions[buf_idx]
529                end = start + row
530                for value in buf[start:end]:
531                    self._dense[out_pos] = elem_dense(int(value))
532                    out_pos += 1
533                positions[buf_idx] = end
534
535        self.set_dense_inplace()

615    def randomize(self, sparsity: float, rng: Optional[random.Random] = None) -> None:
616        """Populate the SDR with random active bits drawn at the requested sparsity."""
617        nta_assert(0.0 <= sparsity <= 1.0, "Sparsity must be within [0, 1].")
618
619        size = int(self.__size)
620        nbits = max(0, min(size, int(round(size * float(sparsity)))))
621        rng = rng or random.Random(0)
622
623        if nbits:
624            selected = sorted(rng.sample(range(size), nbits))
625        else:
626            selected = []
627
628        self._sparse = [elem_sparse(idx) for idx in selected]
629        self.set_sparse_inplace()

631    def add_noise(self, fraction_noise: float, rng: Optional[random.Random] = None) -> None:
632        """Stochastically move active bits while preserving the overall sparsity."""
633        nta_assert(
634            0.0 <= fraction_noise <= 1.0,
635            "Noise fraction must be within [0, 1].",
636        )
637        nta_check(
638            (1.0 + fraction_noise) * self.get_sparsity() <= 1.0,
639            "Noise would exceed SDR capacity.",
640        )
641
642        num_move_bits = int(round(fraction_noise * int(self.get_sum())))
643        if num_move_bits == 0:
644            return
645
646        rng = rng or random.Random()
647
648        sparse_values = list(map(int, self.get_sparse()))
649        nta_assert(
650            len(sparse_values) >= num_move_bits,
651            "Not enough active bits to turn off.",
652        )
653        turn_off = rng.sample(sparse_values, num_move_bits)
654
655        dense = self.get_dense()
656        off_population = [idx for idx in range(int(self.__size)) if int(dense[idx]) == 0]
657        nta_assert(
658            len(off_population) >= num_move_bits,
659            "Not enough inactive bits to turn on.",
660        )
661        turn_on = rng.sample(off_population, num_move_bits)
662
663        for idx in turn_on:
664            dense[idx] = elem_dense(1)
665        for idx in turn_off:
666            dense[idx] = elem_dense(0)
667
668        self.set_dense_inplace()

670    def kill_cells(self, fraction: float, seed: int = 0) -> None:
671        """Deactivate a random subset of bits, seeded for deterministic selection."""
672        nta_check(0.0 <= fraction <= 1.0, "Kill fraction must be within [0, 1].")
673
674        size = int(self.__size)
675        nkill = int(round(size * fraction))
676        if nkill == 0:
677            return
678
679        rng = random.Random(int(seed))
